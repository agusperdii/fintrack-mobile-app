    %% --- ENTITIES ---
    class TransactionType {
        <<enumeration>>
        income
        expense
    }

    class AppData {
        +double initialBalance
        +double totalIncome
        +double totalExpense
        +List~Transaction~ recentTransactions
        +List~AnalysisData~ analysis
        +double? spendingTarget
        +String? targetPeriod
        +double balance
        +fromJson(Map json)$
    }

    class Transaction {
        +String? id
        +String title
        +String? description
        +double amount
        +String category
        +String date
        +TransactionType type
        +String? source
        +String? receiptUrl
        +fromJson(Map json)$
    }

    class AnalysisData {
        +String label
        +double amount
        +String colorHex
        +fromJson(Map json)$
    }

    class NudgeData {
        +String id
        +String type
        +String category
        +String message
        +bool isRead
        +DateTime createdAt
        +fromJson(Map json)$
    }

    class OcrResult {
        +String? storeName
        +double? totalAmount
        +DateTime? date
        +List~OcrItem~ items
        +fromJson(Map json)$
    }

    AppData "1" *-- "many" Transaction
    AppData "1" *-- "many" AnalysisData
    Transaction --> TransactionType

    class RemoteDataSource {

        +getDashboardData()
        +getUserProfile()
        +getTransactions()
        +addTransaction()
        +updateProfile()
        +...()
    }

    class RemoteDataSourceImpl {
        +AuthController authController
        +String baseUrl

        -post(String url, Map body)
        +...()

    RemoteDataSource <|.. RemoteDataSourceImpl

        +...()

    FinanceRepository o-- RemoteDataSource
